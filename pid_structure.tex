\documentclass[tikz, border=10pt]{standalone}

\usetikzlibrary{shapes, arrows, positioning, calc}

% PID controller structure diagram:
% shows the control loop with setpoint (задание), error signal,
% proportional / integral / derivative (P/I/D) controller blocks,
% the controlled process (plant), and the feedback path from the
% process output back to the summing junction.

\begin{document}

\begin{tikzpicture}[
    auto, 
    node distance=2cm, 
    >=latex',
    signal_label/.style={fill=white, inner sep=2pt},
    block/.style={
        draw, 
        fill=white, 
        rectangle, 
        minimum height=1.1cm, 
        minimum width=3.5cm, 
        text width=3.2cm,    
        align=left,           
        inner xsep=5pt        
    },
    process_block/.style={
        block,
        minimum width=2.0cm, 
        text width=1.8cm, 
        align=center
    },
    sum/.style={draw, fill=white, circle, inner sep=2pt}
]

    % Nodes
    
    \node [signal_label] (input_text) {задание};
    \node [sum, right=1.5cm of input_text] (sum1) {\Large $\Sigma$};
    
    % Branch point for distributing error signal to P, I, D blocks
    \node [coordinate, right=1cm of sum1] (branch) {};
    
    % PID controller blocks
    \node [block, right=2.5cm of sum1] (integral) {
        \textbf{I} \hfill $K_i \int_{0}^{t} e(t) dt$
    };
    \node [block, above=0.6cm of integral] (prop) {
        \textbf{P} \hfill $K_p e(t)$
    };
    \node [block, below=0.6cm of integral] (deriv) {
        \textbf{D} \hfill $K_d \frac{de}{dt}$
    };
    
    % Summation node after controller
    \node [sum, right=1.5cm of integral] (sum2) {\Large $\Sigma$};
    \node [process_block, right=1.0cm of sum2] (process) {\textbf{процесс}};
    \node [signal_label, right=0.8cm of process] (out_text) {выход};
    \node [coordinate] (feedback_node) at ($(out_text.east) + (0.3,0)$) {};

    % Connections

    % Input signal
    \draw [->] (input_text) -- node[above, pos=0.9] {$+$} (sum1);
    
    % Error signal and branching to P, I, D
    \node [signal_label] (err_text) at (branch) {ошибка};
    \draw [->] (sum1) -- (err_text);
    
    \draw [->] (err_text) -- (integral);
    \draw [->] (err_text) |- (prop);
    \draw [->] (err_text) |- (deriv);
    
    % Combining signals at sum2
    \draw [->] (prop) -| (sum2);
    \draw [->] (integral) -- (sum2);
    \draw [->] (deriv) -| (sum2);
    
    % From controller to process and to output
    \draw [->] (sum2) -- (process);
    \draw [->] (process) -- (out_text);
    
    % Final output arrow
    \draw (out_text) -- (feedback_node);
    \draw [->] (feedback_node) -- ++(1,0);
    
    % Feedback line
    \draw [->] (feedback_node) |- ++(0,-3.2) -| node[left, pos=0.97] {$-$} (sum1);

\end{tikzpicture}

\end{document}
